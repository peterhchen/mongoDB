# Insert document Syntax:
>db.COLLECTION_NAME.insert(document)

# db.coll.inser ()
>db.mycol.insert({
   _id: ObjectId(7df78ad8902c),
   title: 'MongoDB Overview', 
   description: 'MongoDB is no sql database',
   by: 'tutorials point',
   url: 'http://www.tutorialspoint.com',
   tags: ['mongodb', 'database', 'NoSQL'],
   likes: 100
})

# ID format (total 12 bytes):  
# 4 bytes       3 bytes     2 bytes     3 byte
# timestamp     mchine ID   pID         Increment ID
id: ObjectId(4 bytes timestamp, 3 bytes machine id, 2 bytes process id, 
   3 bytes incrementer)
